106  public function search()

113  return $result;

116  if (!hide($i) || $cf['show_hidden']['pages_search'] == 'true') {
117  $found = true;

119  foreach ($words as $word) {

121  $found = false;
122  break;

124  }

126  $result[] = $i;
127  }
